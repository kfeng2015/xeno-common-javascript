1 /**
  2  * @class
  3  * <p>
  4  * The class is designed for handling URL constructions.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @param baseURL
 13  * 			The base URL without parameters and anchor.
 14  */
 15 xeno.common.lang.URL = function(baseURL) {
 16 	var Data = xeno.common.lang.Data;
 17 
 18 	xc_global_ensure(arguments.length > 0, "Missing required argument 'baseURL', index: 0");
 19 	xc_global_ensure(Data.isString(baseURL), "The argument 'baseURL' is not a string");
 20 
 21 	/**
 22 	 * @description
 23 	 * <p>
 24 	 * Returns the URL parameter by name.
 25 	 * </p>
 26 	 * 
 27 	 * @return The string.
 28 	 */
 29 	this.getParam = function(name) {
 30 		xc_global_ensure(arguments.length > 0, "Missing required argument 'name', index: 0");
 31 		xc_global_ensure(Data.isString(name), "The argument 'name' is not a string");
 32 
 33 		var found = params[name];
 34 
 35 		return Data.isUndefined(found) ? null : found;
 36 	};
 37 
 38 	/**
 39 	 * @description
 40 	 * <p>
 41 	 * Sets the URL parameter, if the named parameter exists, replaced by current setting name-value pair one.
 42 	 * The "Object.toString()" will be used to convert the value to a string.
 43 	 * </p>
 44 	 * 
 45 	 * @param name
 46 	 * 			The name.
 47 	 * @param value
 48 	 * 			The value.
 49 	 */
 50 	this.setParam = function(name, value) {
 51 		xc_global_ensure(arguments.length > 0, "Missing required argument 'name', index: 0");
 52 		xc_global_ensure(arguments.length > 1, "Missing required argument 'value', index: 1");
 53 		xc_global_ensure(Data.isString(name), "The argument 'name' is not a string");
 54 
 55 		params[name] = "" + value;
 56 	};
 57 
 58 	/**
 59 	 * @description
 60 	 * <p>
 61 	 * Removes the URL parameter by name.
 62 	 * </p>
 63 	 * 
 64 	 * @param name
 65 	 * 			The name.
 66 	 */
 67 	this.removeParam = function(name) {
 68 		xc_global_ensure(arguments.length > 0, "Missing required argument 'name', index: 0");
 69 		xc_global_ensure(Data.isString(name), "The argument 'name' is not a string");
 70 
 71 		delete params[name];
 72 	};
 73 
 74 	/**
 75 	 * @description
 76 	 * <p>
 77 	 * Returns the URL anchor.
 78 	 * </p>
 79 	 * 
 80 	 * @return The string.
 81 	 */
 82 	this.getAnchor = function() {
 83 		return anchor;
 84 	};
 85 
 86 	/**
 87 	 * @description
 88 	 * <p>
 89 	 * Sets the URL anchor.
 90 	 * </p>
 91 	 * 
 92 	 * @param name
 93 	 * 			The name.
 94 	 */
 95 	this.setAnchor = function(name) {
 96 		xc_global_ensure(arguments.length > 0, "Missing required argument 'name', index: 0");
 97 		xc_global_ensure(Data.isString(name), "The argument 'name' is not a string");
 98 
 99 		anchor = name;
100 	};
101 
102 	/**
103 	 * @description
104 	 * <p>
105 	 * Removes the URL anchor.
106 	 * </p>
107 	 */
108 	this.removeAnchor = function() {
109 		anchor = null;
110 	};
111 
112 	/**
113 	 * <p>
114 	 * Returns the URL with the string.
115 	 * If any parameters and anchor exist, names and values of them will be encoded by using "encodeURIComponent".
116 	 * </p>
117 	 * 
118 	 * @return The string.
119 	 */
120 	this.toURL = function() {
121 
122 		return buildURL(function(item) {
123 			return encodeURIComponent(item);
124 		});
125 	};
126 
127 	/**
128 	 * @ignore
129 	 */
130 	this.toString = function() {
131 
132 		return buildURL(function(item) {
133 			return item;
134 		});
135 	};
136 
137 	/**
138 	 * @ignore
139 	 */
140 	var buildURL = function(callback) {
141 		var queryParams = [];
142 		var queryAnchor = "";
143 
144 		for(var i in params) {
145 			queryParams.push(callback(i) + "=" + callback(params[i]));
146 		}
147 
148 		if(anchor !== null) {
149 			queryAnchor = callback(anchor);
150 		}
151 
152 		var pageParams = queryParams.length > 0 ? ("?" + queryParams.join("&")) : "";
153 		var pageAnchor = queryAnchor.length > 0 ? ("#" + queryAnchor) : "";
154 
155 		return baseURL + pageParams + pageAnchor;
156 	};
157 
158 	var params = {};
159 	var anchor = null;
160 };
