1 /**
  2  * @class
  3  * <p>
  4  * A set of utilities for random operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  */
 12 xeno.common.util.RandomUtils = (function () {
 13 	var NumberUtils = null;
 14 
 15 	xeno.common.onCompleteCallbacks.push(function() {
 16 		NumberUtils = xeno.common.util.NumberUtils;
 17 	});
 18 
 19 	return {
 20 
 21 		/**
 22 		 * @description
 23 		 * <p>
 24 		 * Generates and returns an integer number between (inclusive) the "min" and "max" in randomly.
 25 		 * </p>
 26 		 * 
 27 		 * @param min
 28 		 * 			The integer minimum value.
 29 		 * @param max
 30 		 * 			The integer maximum value.
 31 		 * 
 32 		 * @return
 33 		 * 			The integer number.
 34 		 */
 35 		generateInt: function(min, max) {
 36 			xc_global_ensure(arguments.length > 0, "Missing required argument 'min', index: 0");
 37 			xc_global_ensure(arguments.length > 1, "Missing required argument 'max', index: 1");
 38 			xc_global_ensure(NumberUtils.isInt(min), "The argument 'min' is not an integer number");
 39 			xc_global_ensure(NumberUtils.isInt(max), "The argument 'max' is not an integer number");
 40 			xc_global_ensure(min <= max, "The argument 'min' is greater than the argument 'max'");
 41 
 42 			return Math.floor(Math.random() * (max - min + 1)) + min;
 43 		}
 44 	};
 45 
 46 })();
