1 /**
  2  * @class
  3  * <p>
  4  * The class is designed for converting a number into a readable format by the pattern provides.
  5  * The number could be rounded to the specified number of decimal places by the "pattern" provides, the "Number.toFixed(n)" action will be processed.
  6  * </p>
  7  * 
  8  * @description
  9  * <p>
 10  * The class constructor.
 11  * </p>
 12  * 
 13  * @example
 14  * <b>Examples</b>
 15  * 
 16  * new xeno.common.text.NumberFormat("00000.##").format(1000.2) = "01000.2"
 17  * new xeno.common.text.NumberFormat("00.00").format(1.235) = "01.24"
 18  * 
 19  * @example
 20  * <b>Patterns</b>
 21  * 
 22  * 0: digit
 23  * #: digit, zero shows as absent
 24  * .: decimal separator (should appear only once and not start or end with)
 25  * 
 26  * @param pattern
 27  * 			The pattern to be used.
 28  */
 29 xeno.common.text.NumberFormat = function(pattern) {
 30 	var Data = xeno.common.lang.Data;
 31 	var StringUtils = xeno.common.util.StringUtils;
 32 
 33 	xc_global_ensure(arguments.length > 0, "Missing required argument 'pattern', index: 0");
 34 	xc_global_ensure(Data.isString(pattern), "The argument 'pattern' is not a string");
 35 	xc_global_ensure(/^((0+)|(#+0*))(\.((#+)|(0+#*)))?$/.test(pattern), "Invalid number pattern: " + pattern);
 36 
 37 	/**
 38 	 * @description
 39 	 * <p>
 40 	 * Formats the number.
 41 	 * </p>
 42 	 * 
 43 	 * @param num
 44 	 * 			The number to be formatted.
 45 	 * 
 46 	 * @return
 47 	 * 			The string.
 48 	 */
 49 	this.format = function(num) {
 50 		xc_global_ensure(arguments.length > 0, "Missing required argument 'num', index: 0");
 51 		xc_global_ensure(Data.isNumber(num), "The argument 'num' is not a number");
 52 
 53 		var currentItems = StringUtils.removeStart(num.toFixed(decimalMasks.length), "-").split(".");
 54 		var integerItems = currentItems[0].split("").reverse();
 55 		var decimalItems = currentItems.length > 1 ? currentItems[1].split("") : [];
 56 
 57 		var integerParts = [];
 58 		var decimalParts = [];
 59 
 60 		if(integerItems.length >= integerMasks.length) {
 61 			integerParts = integerItems.concat([]);
 62 
 63 		} else {
 64 			match(integerMasks, integerItems, integerParts);
 65 		}
 66 
 67 		match(decimalMasks, decimalItems, decimalParts);
 68 
 69 		var integerValue = integerParts.reverse().join("");
 70 		var decimalValue = StringUtils.removeEnd("." + decimalParts.join(""), ".");
 71 
 72 		return (num < 0 ? "-" : "") + integerValue + decimalValue;
 73 	};
 74 
 75 	/**
 76 	 * @ignore
 77 	 */
 78 	var match = function(masks, items, parts) {
 79 
 80 		for(var i = 0; i < masks.length; i += 1) {
 81 			var mask = masks[i];
 82 			var item = items[i];
 83 			var pass = Data.isUndefined(item) || item === "0";
 84 
 85 			if(mask === "#" && pass) {
 86 				parts.push("");
 87 
 88 			} else if(mask === "0" && pass) {
 89 				parts.push("0");
 90 
 91 			} else {
 92 				parts.push(item);
 93 			}
 94 		}
 95 	};
 96 
 97 	var patternMasks = pattern.split(".");
 98 	var integerMasks = patternMasks[0].split("").reverse();
 99 	var decimalMasks = patternMasks.length > 1 ? patternMasks[1].split("") : [];
100 };
