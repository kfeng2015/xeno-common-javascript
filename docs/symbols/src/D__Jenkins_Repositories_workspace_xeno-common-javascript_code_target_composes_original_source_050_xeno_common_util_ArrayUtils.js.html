1 /**
  2  * @class
  3  * <p>
  4  * A set of utilities for array operations.
  5  * </p>
  6  * <p>
  7  * When you use this class for some certain functions which have return values, a new array instance will be created and return.
  8  * The original array passes in will not be edited, but elements references in the new array will be the same as their in the original array.
  9  * </p>
 10  * 
 11  * @description
 12  * <p>
 13  * The class constructor.
 14  * </p>
 15  */
 16 xeno.common.util.ArrayUtils = (function () {
 17 	var Data = null;
 18 	var NumberUtils = null;
 19 
 20 	xeno.common.onCompleteCallbacks.push(function() {
 21 		Data = xeno.common.lang.Data;
 22 		NumberUtils = xeno.common.util.NumberUtils;
 23 	});
 24 
 25 	return {
 26 
 27 		/**
 28 		 * @description
 29 		 * <p>
 30 		 * Checks whether the "arr" contains an "item" or not.
 31 		 * </p>
 32 		 * <p>
 33 		 * The rule of compare data's equal in the array is the same to JavaScript.
 34 		 * Complex objects will compare their references, while others like number or string and so on, will just compare their values.
 35 		 * </p>
 36 		 * 
 37 		 * @example
 38 		 * <b>Examples</b>
 39 		 * 
 40 		 * Assumes the "obj" is: { id: 1, name: "eric" }
 41 		 * Assumes the "newObj" is: { id: 1, name: "eric" }
 42 		 * Assumes the "arr" is: ["1", null, undefined, obj, "5", undefined, 6, true, obj]
 43 		 * 
 44 		 * xeno.common.util.ArrayUtils.contains(arr, "11") = false
 45 		 * xeno.common.util.ArrayUtils.contains(arr, "1") = true
 46 		 * xeno.common.util.ArrayUtils.contains(arr, null) = true
 47 		 * xeno.common.util.ArrayUtils.contains(arr, undefined) = true
 48 		 * xeno.common.util.ArrayUtils.contains(arr, obj) = true
 49 		 * xeno.common.util.ArrayUtils.contains(arr, newObj) = false
 50 		 * 
 51 		 * @param arr
 52 		 * 			The array to be checked.
 53 		 * @param item
 54 		 * 			The item in the array to be found.
 55 		 * 
 56 		 * @return
 57 		 * 			The boolean.
 58 		 */
 59 		contains: function(arr, item) {
 60 			xc_global_ensure(arguments.length > 0, "Missing required argument 'arr', index: 0");
 61 			xc_global_ensure(arguments.length > 1, "Missing required argument 'item', index: 1");
 62 			xc_global_ensure(Data.isArray(arr), "The argument 'arr' is not an array");
 63 
 64 			return arr.indexOf(item) >= 0;
 65 		},
 66 
 67 		/**
 68 		 * @description
 69 		 * <p>
 70 		 * Inserts a certain "item" into the "arr" and returns it with a new array instance.
 71 		 * </p>
 72 		 * 
 73 		 * @example
 74 		 * <b>Examples</b>
 75 		 * 
 76 		 * Assumes the "obj" is: { id: 1, name: "eric" }
 77 		 * Assumes the "arr" is: ["1", "2", "3", "4", "5"]
 78 		 * 
 79 		 * xeno.common.util.ArrayUtils.insert(arr, "6") = ["1", "2", "3", "4", "5", "6"]
 80 		 * xeno.common.util.ArrayUtils.insert(arr, "6", 0) = ["6", "1", "2", "3", "4", "5"]
 81 		 * xeno.common.util.ArrayUtils.insert(arr, obj, 3) = ["1", "2", "3", obj, "4", "5"]
 82 		 * xeno.common.util.ArrayUtils.insert(arr, "6", 5) = ["1", "2", "3", "4", "5", "6"]
 83 		 * xeno.common.util.ArrayUtils.insert(arr, "6", 6) = ["1", "2", "3", "4", "5", "6"]
 84 		 * 
 85 		 * @param arr
 86 		 * 			The array to be handled.
 87 		 * @param item
 88 		 * 			The item to be inserted into.
 89 		 * @param index
 90 		 * 			The position to be inserted at, optional.<br/>
 91 		 * 			If this value is out of the array's range, the "item" will be inserted at the last position.<br/>
 92 		 * 			The default value is: Number.MAX_VALUE.
 93 		 * 
 94 		 * @return
 95 		 * 			The array.
 96 		 */
 97 		insert: function(arr, item, index) {
 98 			xc_global_ensure(arguments.length > 0, "Missing required argument 'arr', index: 0");
 99 			xc_global_ensure(arguments.length > 1, "Missing required argument 'item', index: 1");
100 			xc_global_ensure(Data.isArray(arr), "The argument 'arr' is not an array");
101 
102 			var fixedIndex = Number.MAX_VALUE;
103 
104 			if(arguments.length > 2) {
105 				xc_global_ensure(NumberUtils.isInt(index), "The argument 'index' is not an integer number");
106 				xc_global_ensure(index >= 0, "The argument 'index' is negative");
107 
108 				fixedIndex = index;
109 			}
110 
111 			var items = arr.concat([]);
112 			items.splice(fixedIndex, 0, item);
113 
114 			return items;
115 		},
116 
117 		/**
118 		 * @description
119 		 * <p>
120 		 * Removes a certain data from the "arr" by its index in it and returns a new array instance.
121 		 * </p>
122 		 * <p>
123 		 * This function also safe handling the "index" passes in which is out of range, it will not take effects and return the same array with different reference according to the "arr".
124 		 * </p>
125 		 * 
126 		 * @example
127 		 * <b>Examples</b>
128 		 * 
129 		 * xeno.common.util.ArrayUtils.remove(["a", 7, false], 1) = ["a", false]
130 		 * xeno.common.util.ArrayUtils.remove(["a", 7, false], 3) = ["a", 7, false]
131 		 * 
132 		 * @param arr
133 		 * 			The array to be handled.
134 		 * @param index
135 		 * 			The position to be removed at.
136 		 * 
137 		 * @return
138 		 * 			The array.
139 		 */
140 		remove: function(arr, index) {
141 			xc_global_ensure(arguments.length > 0, "Missing required argument 'arr', index: 0");
142 			xc_global_ensure(arguments.length > 1, "Missing required argument 'index', index: 1");
143 			xc_global_ensure(Data.isArray(arr), "The argument 'arr' is not an array");
144 			xc_global_ensure(NumberUtils.isInt(index), "The argument 'index' is not an integer number");
145 			xc_global_ensure(index >= 0, "The argument 'index' is negative");
146 
147 			var items = arr.concat([]);
148 			items.splice(index, 1);
149 
150 			return items;
151 		},
152 
153 		/**
154 		 * @description
155 		 * <p>
156 		 * Returns a new sub array from the "arr", the returned sub array starts with the data in the begin position and ends before the end position, and all position counting is zero-based.
157 		 * </p>
158 		 * 
159 		 * @example
160 		 * <b>Examples</b>
161 		 * 
162 		 * Assumes the "arr" is: [1, 2, 3, 4]
163 		 * 
164 		 * xeno.common.util.ArrayUtils.subarray([], 0) = []
165 		 * xeno.common.util.ArrayUtils.subarray([], 1) = []
166 		 * xeno.common.util.ArrayUtils.subarray(arr, 0) = [1, 2, 3, 4]
167 		 * xeno.common.util.ArrayUtils.subarray(arr, 2) = [3, 4]
168 		 * xeno.common.util.ArrayUtils.subarray(arr, 3) = [4]
169 		 * xeno.common.util.ArrayUtils.subarray(arr, 4) = []
170 		 * xeno.common.util.ArrayUtils.subarray(arr, 1, 3) = [2, 3]
171 		 * xeno.common.util.ArrayUtils.subarray(arr, 1, 4) = [2, 3, 4]
172 		 * xeno.common.util.ArrayUtils.subarray(arr, 1, 5) = [2, 3, 4]
173 		 * 
174 		 * @param arr
175 		 * 			The array to get the sub array from.
176 		 * @param beginIndex
177 		 * 			The position to start from.
178 		 * @param endIndex
179 		 * 			The position to end at (exclusive), optional.<br/>
180 		 * 			The default value is: Number.MAX_VALUE.
181 		 * 
182 		 * @return
183 		 * 			The array.
184 		 */
185 		subarray: function(arr, beginIndex, endIndex) {
186 			xc_global_ensure(arguments.length > 0, "Missing required argument 'arr', index: 0");
187 			xc_global_ensure(arguments.length > 1, "Missing required argument 'beginIndex', index: 1");
188 			xc_global_ensure(Data.isArray(arr), "The argument 'arr' is not an array");
189 			xc_global_ensure(NumberUtils.isInt(beginIndex), "The argument 'beginIndex' is not an integer number");
190 			xc_global_ensure(beginIndex >= 0, "The argument 'beginIndex' is negative");
191 
192 			var fixedEndIndex = Number.MAX_VALUE;
193 
194 			if(arguments.length > 2) {
195 				xc_global_ensure(NumberUtils.isInt(endIndex), "The argument 'endIndex' is not an integer number");
196 				xc_global_ensure(beginIndex <= endIndex, "The argument 'beginIndex' is greater than the argument 'endIndex'");
197 
198 				fixedEndIndex = endIndex;
199 			}
200 
201 			return arr.concat([]).slice(beginIndex, fixedEndIndex);
202 		}
203 
204 	};
205 
206 })();
