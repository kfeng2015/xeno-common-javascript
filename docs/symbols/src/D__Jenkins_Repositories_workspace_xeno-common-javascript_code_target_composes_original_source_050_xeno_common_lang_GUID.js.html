 12 xeno.common.lang.GUID = (function () {
 13 
 14 	return {
 15 
 16 		/**
 17 		 * @description
 18 		 * <p>
 19 		 * Generates and returns a GUID value in "########-####-4###-Y###-############" (lowercase) format.
 20 		 * </p>
 21 		 * <ul>
 22 		 * 	<li>Bits 12-15 of the "time_hi_and_version" field to 0010.</li>
 23 		 * 	<li>Bits 6-7 of the "clock_seq_hi_and_reserved" field to 01.</li>
 24 		 * </ul>
 25 		 * 
 26 		 * @return
 27 		 * 			The string.
 28 		 */
 29 		nextValue: function() {
 30 
 31 			return "xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx".replace(/[xy]/g, function(c) {
 32 				var r = Math.random() * 16 | 0;
 33 				var v = (c === "x") ? r : (r & 0x3 | 0x8);
 34 
 35 				return v.toString(16);
 36 			});
 37 		}
 38 
 39 	};
 40 
 41 })();
