1 /**
  2  * @class
  3  * <p>
  4  * A set of utilities for date operations.
  5  * </p>
  6  * <p>
  7  * When you use this class for some certain functions which have return values, a new date instance will be created and return.
  8  * The original date passes in will not be edited.
  9  * </p>
 10  * 
 11  * @description
 12  * <p>
 13  * The class constructor.
 14  * </p>
 15  */
 16 xeno.common.util.DateUtils = (function () {
 17 	var Data = null;
 18 	var DateUtils = null;
 19 	var NumberUtils = null;
 20 
 21 	xeno.common.onCompleteCallbacks.push(function() {
 22 		Data = xeno.common.lang.Data;
 23 		DateUtils = xeno.common.util.DateUtils;
 24 		NumberUtils = xeno.common.util.NumberUtils;
 25 	});
 26 
 27 	var MILLISECOND_OF_DAY = 86400000;
 28 
 29 	return {
 30 
 31 		/**
 32 		 * @description
 33 		 * <p>
 34 		 * Tests whether the "year" is a leap year or not.
 35 		 * </p>
 36 		 * 
 37 		 * @param year
 38 		 * 			The year to be tested.
 39 		 * 
 40 		 * @return
 41 		 * 			The boolean.
 42 		 */
 43 		isLeapYear: function(year) {
 44 			xc_global_ensure(arguments.length > 0, "Missing required argument 'year', index: 0");
 45 			xc_global_ensure(NumberUtils.isInt(year), "The argument 'year' is not an integer number");
 46 			xc_global_ensure(year >= 0, "The argument 'year' is negative");
 47 
 48 			if(year % 400 === 0) {
 49 				return true;
 50 			}
 51 
 52 			if(year % 4 === 0 && year % 100 !== 0) {
 53 				return true;
 54 			}
 55 
 56 			return false;
 57 		},
 58 
 59 		/**
 60 		 * @description
 61 		 * <p>
 62 		 * Returns the day sequence of the year.
 63 		 * </p>
 64 		 * 
 65 		 * @param date
 66 		 * 			The date.
 67 		 * 
 68 		 * @return
 69 		 * 			The number.
 70 		 */
 71 		getDayOfYear: function(date) {
 72 			xc_global_ensure(arguments.length > 0, "Missing required argument 'date', index: 0");
 73 			xc_global_ensure(Data.isDate(date), "The argument 'date' is not a date");
 74 
 75 			var fixedDate = new Date(date.getFullYear(), date.getMonth(), date.getDate());
 76 			var firstDate = new Date(date.getFullYear(), 0, 1);
 77 
 78 			return (fixedDate.getTime() - firstDate.getTime()) / MILLISECOND_OF_DAY + 1;
 79 		},
 80 
 81 		/**
 82 		 * @description
 83 		 * <p>
 84 		 * Returns the day sequence of the month.
 85 		 * </p>
 86 		 * 
 87 		 * @param date
 88 		 * 			The date.
 89 		 * 
 90 		 * @return
 91 		 * 			The number.
 92 		 */
 93 		getDayOfMonth: function(date) {
 94 			xc_global_ensure(arguments.length > 0, "Missing required argument 'date', index: 0");
 95 			xc_global_ensure(Data.isDate(date), "The argument 'date' is not a date");
 96 
 97 			return date.getDate();
 98 		},
 99 
100 		/**
101 		 * @description
102 		 * <p>
103 		 * Returns the day sequence of the week according to ISO-8601 definition.
104 		 * </p>
105 		 * <ul>
106 		 * 	<li>1: Monday</li>
107 		 * 	<li>2: Tuesday</li>
108 		 * 	<li>3: Wednesday</li>
109 		 * 	<li>4: Thursday</li>
110 		 * 	<li>5: Friday</li>
111 		 * 	<li>6: Saturday</li>
112 		 * 	<li>7: Sunday</li>
113 		 * </ul>
114 		 * 
115 		 * @param date
116 		 * 			The date.
117 		 * 
118 		 * @return
119 		 * 			The number.
120 		 */
121 		getDayOfWeek: function(date) {
122 			xc_global_ensure(arguments.length > 0, "Missing required argument 'date', index: 0");
123 			xc_global_ensure(Data.isDate(date), "The argument 'date' is not a date");
124 
125 			// return date.getDay() || 7;
126 
127 			var day = date.getDay();
128 
129 			return day === 0 ? 7 : day;
130 		},
131 
132 		/**
133 		 * @description
134 		 * <p>
135 		 * Returns the month sequence of the year.
136 		 * </p>
137 		 * <ul>
138 		 * 	<li>1: January</li>
139 		 * 	<li>2: February</li>
140 		 * 	<li>3: March</li>
141 		 * 	<li>4: April</li>
142 		 * 	<li>5: May</li>
143 		 * 	<li>6: June</li>
144 		 * 	<li>7: July</li>
145 		 * 	<li>8: August</li>
146 		 * 	<li>9: September</li>
147 		 * 	<li>10: October</li>
148 		 * 	<li>11: November</li>
149 		 * 	<li>12: December</li>
150 		 * </ul>
151 		 * 
152 		 * @param date
153 		 * 			The date.
154 		 * 
155 		 * @return
156 		 * 			The number.
157 		 */
158 		getMonthOfYear: function(date) {
159 			xc_global_ensure(arguments.length > 0, "Missing required argument 'date', index: 0");
160 			xc_global_ensure(Data.isDate(date), "The argument 'date' is not a date");
161 
162 			return date.getMonth() + 1;
163 		},
164 
165 		/**
166 		 * @description
167 		 * <p>
168 		 * Returns the week sequence of the year.
169 		 * The default calculation follows the ISO-8601 definition - the week starts on Monday, the first week of the year contains the first Thursday of the year.
170 		 * This means that some days from one year may be placed into weeks "belonging" to another year.
171 		 * </p>
172 		 * 
173 		 * @param date
174 		 * 			The date.
175 		 * 
176 		 * @return
177 		 * 			The number.
178 		 */
179 		getWeekOfYear: function(date) {
180 			xc_global_ensure(arguments.length > 0, "Missing required argument 'date', index: 0");
181 			xc_global_ensure(Data.isDate(date), "The argument 'date' is not a date");
182 
183 			var fixedDate = new Date(date.getFullYear(), date.getMonth(), date.getDate() + 4 - DateUtils.getDayOfWeek(date));
184 			var firstDate = new Date(fixedDate.getFullYear(), 0, 1);
185 
186 			return Math.ceil((((fixedDate.getTime() - firstDate.getTime()) / MILLISECOND_OF_DAY) + 1) / 7);
187 		}
188 
189 	};
190 
191 })();
