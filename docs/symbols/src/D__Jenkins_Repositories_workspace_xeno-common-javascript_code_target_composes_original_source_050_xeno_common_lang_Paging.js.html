1 /**
  2  * @class
  3  * <p>
  4  * The class is designed for handling pagination calculations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @param dataTotal
 13  * 			The total data count.
 14  * @param pageSize
 15  * 			The page size.
 16  * @param pageNum
 17  * 			The current page number.
 18  */
 19 xeno.common.lang.Paging = function(dataTotal, pageSize, pageNum) {
 20 	var NumberUtils = xeno.common.util.NumberUtils;
 21 
 22 	xc_global_ensure(arguments.length > 0, "Missing required argument 'dataTotal', index: 0");
 23 	xc_global_ensure(arguments.length > 1, "Missing required argument 'pageSize', index: 1");
 24 	xc_global_ensure(arguments.length > 2, "Missing required argument 'pageNum', index: 2");
 25 	xc_global_ensure(NumberUtils.isInt(dataTotal), "The argument 'dataTotal' is not an integer number");
 26 	xc_global_ensure(NumberUtils.isInt(pageSize), "The argument 'pageSize' is not an integer number");
 27 	xc_global_ensure(NumberUtils.isInt(pageNum), "The argument 'pageNum' is not an integer number");
 28 
 29 	/**
 30 	 * @description
 31 	 * <p>
 32 	 * Returns the current page number.
 33 	 * </p>
 34 	 * 
 35 	 * @return The number.
 36 	 */
 37 	this.getPageNum = function() {
 38 		return pageNum;
 39 	}
 40 
 41 	/**
 42 	 * @description
 43 	 * <p>
 44 	 * Returns the total page count.
 45 	 * </p>
 46 	 * 
 47 	 * @return The number.
 48 	 */
 49 	this.getPageTotal = function() {
 50 		return pageTotal;
 51 	}
 52 
 53 	/**
 54 	 * @description
 55 	 * <p>
 56 	 * Returns the page size.
 57 	 * </p>
 58 	 * 
 59 	 * @return The number.
 60 	 */
 61 	this.getPageSize = function() {
 62 		return pageSize;
 63 	}
 64 
 65 	/**
 66 	 * @description
 67 	 * <p>
 68 	 * Returns the total data count.
 69 	 * </p>
 70 	 * 
 71 	 * @return The number.
 72 	 */
 73 	this.getDataTotal = function() {
 74 		return dataTotal;
 75 	}
 76 
 77 	/**
 78 	 * @ignore
 79 	 */
 80 	var calculatePageTotal = function(dataTotal, pageSize) {
 81 
 82 		if (dataTotal === MIN_DATA_TOTAL) {
 83 			return MIN_PAGE_NUM;
 84 		}
 85 
 86 		return Math.ceil(dataTotal / pageSize);
 87 	};
 88 
 89 	/**
 90 	 * @ignore
 91 	 */
 92 	var calculatePageNum = function(pageTotal, pageNum) {
 93 
 94 		if (pageTotal === MIN_PAGE_NUM) {
 95 			return MIN_PAGE_NUM;
 96 		}
 97 
 98 		if (pageNum <= MIN_PAGE_NUM) {
 99 			return MIN_PAGE_NUM + 1;
100 		}
101 
102 		if (pageNum >= pageTotal) {
103 			return pageTotal;
104 		}
105 
106 		return pageNum;
107 	};
108 
109 	var MIN_PAGE_NUM = 0;
110 	var MIN_PAGE_SIZE = 1;
111 	var MIN_DATA_TOTAL = 0;
112 
113 	var dataTotal = dataTotal < MIN_DATA_TOTAL ? MIN_DATA_TOTAL : dataTotal;
114 	var pageSize = pageSize < MIN_PAGE_SIZE ? MIN_PAGE_SIZE : pageSize;
115 	var pageTotal = calculatePageTotal(dataTotal, pageSize);
116 	var pageNum = calculatePageNum(pageTotal, pageNum);
117 };
