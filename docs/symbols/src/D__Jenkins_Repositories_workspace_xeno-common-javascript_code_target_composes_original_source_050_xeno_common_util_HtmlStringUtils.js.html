1 /**
  2  * @class
  3  * <p>
  4  * A set of utilities for HTML string operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  */
 12 xeno.common.util.HtmlStringUtils = (function () {
 13 	var Data = null;
 14 
 15 	xeno.common.onCompleteCallbacks.push(function() {
 16 		Data = xeno.common.lang.Data;
 17 	});
 18 
 19 	return {
 20 
 21 		/**
 22 		 * @description
 23 		 * <p>
 24 		 * Encodes the HTML string to the normal string.
 25 		 * </p>
 26 		 * 
 27 		 * @example
 28 		 * <b>Examples</b>
 29 		 * 
 30 		 * xeno.common.util.HtmlStringUtils.encode("&") = "&"
 31 		 * xeno.common.util.HtmlStringUtils.encode("'") = "'"
 32 		 * xeno.common.util.HtmlStringUtils.encode("a<b>b</b>c") = "a<b>b</b>c"
 33 		 * 
 34 		 * @param str
 35 		 * 			The string to be encoded.
 36 		 * 
 37 		 * @return
 38 		 * 			The string.
 39 		 */
 40 		encode: function(str) {
 41 			xc_global_ensure(arguments.length > 0, "Missing required argument 'str', index: 0");
 42 			xc_global_ensure(Data.isString(str), "The argument 'str' is not a string");
 43 
 44 			return str.replace(/[<>&'"\s]/g, function(c) {
 45 
 46 				return {
 47 					"<": "<",
 48 					">": ">",
 49 					"&": "&",
 50 					"'": "'",
 51 					"\"": """,
 52 					" ": " "
 53 				}[c];
 54 			});
 55 		},
 56 
 57 		/**
 58 		 * @description
 59 		 * <p>
 60 		 * Decodes the normal string to the HTML string.
 61 		 * </p>
 62 		 * 
 63 		 * @example
 64 		 * <b>Examples</b>
 65 		 * 
 66 		 * xeno.common.util.HtmlStringUtils.decode("&") = "&"
 67 		 * xeno.common.util.HtmlStringUtils.decode("'") = "'"
 68 		 * xeno.common.util.HtmlStringUtils.decode("a<b>b</b>c") = "a<b>b</b>c"
 69 		 * 
 70 		 * @param str
 71 		 * 			The string to be decoded.
 72 		 * 
 73 		 * @return
 74 		 * 			The string.
 75 		 */
 76 		decode: function(str) {
 77 			xc_global_ensure(arguments.length > 0, "Missing required argument 'str', index: 0");
 78 			xc_global_ensure(Data.isString(str), "The argument 'str' is not a string");
 79 
 80 			return str.replace(/(<)|(>)|(&)|(')|(")|( )/g, function(w) {
 81 
 82 				return {
 83 					"<": "<",
 84 					">": ">",
 85 					"&": "&",
 86 					"'": "'",
 87 					""": "\"",
 88 					" ": " "
 89 				}[w];
 90 			});
 91 		}
 92 
 93 	};
 94 
 95 })();
